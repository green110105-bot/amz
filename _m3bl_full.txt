function adsReq(method, path, { body, multipart, qs, token = TOKEN, storeId = STORE_ID } = {}) {
  const url = 'http://localhost' + path + (qs ? '?' + new URLSearchParams(qs).toString() : '');
  const headers = { 'authorization': 'Bearer ' + token, 'x-store-id': storeId };
  let init = { method, headers };
  if (multipart) {
    const boundary = '----amzqa' + randomBytes(6).toString('hex');
    headers['content-type'] = `multipart/form-data; boundary=${boundary}`;
    const parts = [];
    for (const [name, val] of Object.entries(multipart)) {
      parts.push(`--${boundary}\r\n`);
      if (val && typeof val === 'object' && val.filename != null) {
        parts.push(`Content-Disposition: form-data; name="${name}"; filename="${val.filename}"\r\n\r\n`);
        parts.push(val.content);
        parts.push('\r\n');
      } else {
        parts.push(`Content-Disposition: form-data; name="${name}"\r\n\r\n`);
        parts.push(String(val));
        parts.push('\r\n');
      }
    }
    parts.push(`--${boundary}--\r\n`);
    init.body = Buffer.from(parts.join(''));
  } else if (body !== undefined) {
    headers['content-type'] = 'application/json';
    init.body = JSON.stringify(body);
  }
  return new Request(url, init);
}

async function callAds(method, path, opts) {
  const res = await handleAdsRequest(adsReq(method, path, opts));
  assert.ok(res, 'route returned null for ' + method + ' ' + path);
  const text = await res.text();
  let json = null;
  try { json = text ? JSON.parse(text) : null; } catch {}
  return { status: res.status, body: json };
}
async function callStore(method, path, opts) {
  const res = await handleStoreRequest(adsReq(method, path, opts));
  assert.ok(res, 'store route returned null for ' + method + ' ' + path);
  const text = await res.text();
  let json = null;
  try { json = text ? JSON.parse(text) : null; } catch {}
  return { status: res.status, body: json };
}

async function assertQueueDryRunRoundTrip(item) {
  // M3-P0-03/04: LX manual writes are needs_review and CANNOT be approved
  // (approve_requires_passed_guardrail). They also cannot execute without
  // approval — assert that block first, then dry-run via the explicit force
  // path (operator override) so the run still records dry_run + revert.
  const approveAttempt = await callAds('POST', '/api/v1/store/ads/action-queue/' + item.id + '/approve', { body: { note: 'qa approve attempt' } });
  assert.equal(approveAttempt.status, 400);
  assert.match(approveAttempt.body.message, /approve_requires_passed_guardrail/);
  const blocked = await callAds('POST', '/api/v1/store/ads/action-queue/' + item.id + '/execute', { body: {} });
  assert.equal(blocked.status, 400);
  assert.match(blocked.body.message, /approval_required/);
  const executed = await callAds('POST', '/api/v1/store/ads/action-queue/' + item.id + '/execute', { body: { force: true } });
  assert.equal(executed.status, 200);
  assert.equal(executed.body.state, 'executed');
  assert.equal(executed.body.latestRun.dryRun, true);
  const reverted = await callAds('POST', '/api/v1/store/ads/action-queue/' + item.id + '/revert', { body: { reason: 'qa revert queued lx write' } });
  assert.equal(reverted.status, 200);
  assert.equal(reverted.body.state, 'reverted');
  return reverted.body;
}

// M3-P0-03/M3-P0-04: approve is gated on a passing guardrail. Seed a suggestion
// whose server-recomputed guardrail is 'passed' (ADJUST_BID, confident, evidence
// present) so the approve→execute→revert happy path is exercised honestly.
function seedPassedSuggestion(id) {
  const db = getDbInstance();
  const now = new Date().toISOString();
  db.prepare(`INSERT INTO ad_suggestions(id,user_id,store_id,time_bucket,severity,action_type,entity,title,summary,detail,confidence,evidence,state,created_at) VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?)`).run(
    id, USER_ID, STORE_ID, now, JSON.stringify({ level: 'P2' }), JSON.stringify({ primitive: 'ADJUST_BID', label: '调竞价' }),
    JSON.stringify({ kind: 'keyword', keyword: 'generic widget ' + id, campaignId: 'c1', adGroupId: 'g1', targetingId: 't1' }),
    'qa passed', 's', 'd', 0.82,
    JSON.stringify([{ label: 'acos', value: '30%', signal: 'warn' }, { label: 'spend', value: '$100', signal: 'info' }]),
    'pending', now,
  );
  return id;
}

test('M3-aq-01: action queue add/approve/execute/revert dry-run loop', async () => {
  const sugId = seedPassedSuggestion('sug-qa-aq01');
  const sug = (await callAds('GET', '/api/v1/store/ads/suggestions/' + sugId)).body;
  assert.equal(sug.guardrail?.status, 'passed', 'seeded suggestion should have a passed guardrail');

  const enq = await callAds('POST', '/api/v1/store/ads/suggestions/' + sug.id + '/enqueue', { body: { note: 'qa queue' } });
  assert.equal(enq.status, 201);
  assert.equal(enq.body.suggestionId, sug.id);
  assert.equal(enq.body.state, 'queued');
  assert.ok(enq.body.typedAction?.actionPrimitive);
  assert.equal(enq.body.guardrail?.status, 'passed');

  const dup = await callAds('POST', '/api/v1/store/ads/action-queue', { body: { suggestionId: sug.id } });
  assert.equal(dup.status, 200);
  assert.equal(dup.body.duplicate, true);

  const approved = await callAds('POST', '/api/v1/store/ads/action-queue/' + enq.body.id + '/approve', { body: { note: 'qa approve' } });
  assert.equal(approved.status, 200);
  assert.equal(approved.body.state, 'approved');
  assert.ok(approved.body.approvedAt);

  const executed = await callAds('POST', '/api/v1/store/ads/action-queue/' + enq.body.id + '/execute', { body: {} });
  assert.equal(executed.status, 200);
  assert.equal(executed.body.state, 'executed');
  assert.equal(executed.body.latestRun.status, 'dry_run_success');
  assert.equal(executed.body.latestRun.dryRun, true);

  const freshSug = await callAds('GET', '/api/v1/store/ads/suggestions/' + sug.id);
  assert.equal(freshSug.body.state, 'observing');

  const runs = await callAds('GET', '/api/v1/store/ads/action-runs', { qs: { queueItemId: enq.body.id } });
  assert.equal(runs.status, 200);
  assert.ok(runs.body.items.some((x) => x.status === 'dry_run_success'));

  const reverted = await callAds('POST', '/api/v1/store/ads/action-queue/' + enq.body.id + '/revert', { body: { reason: 'qa revert action queue' } });
  assert.equal(reverted.status, 200);
  assert.equal(reverted.body.state, 'reverted');

  const revertedSug = await callAds('GET', '/api/v1/store/ads/suggestions/' + sug.id);
  assert.equal(revertedSug.body.state, 'pending');
});

// M3-P0-02: batch approve must skip (never auto-approve) needs_review items.
test('M3-P0-02: batch{approve:true} skips needs_review, executes passed', async () => {
  const passedId = seedPassedSuggestion('sug-qa-p002-pass');
  const list = await callAds('GET', '/api/v1/store/ads/suggestions', { qs: { state: 'pending' } });
  const needsReview = list.body.items.find((x) => x.guardrail?.status === 'needs_review');
  assert.ok(needsReview?.id, 'need a needs_review suggestion');

  const enqPassed = await callAds('POST', '/api/v1/store/ads/action-queue', { body: { suggestionId: passedId } });
  const enqNeeds = await callAds('POST', '/api/v1/store/ads/action-queue', { body: { suggestionId: needsReview.id } });
  assert.equal(enqNeeds.body.guardrail.status, 'needs_review');

  const batch = await callAds('POST', '/api/v1/store/ads/action-queue/execute-batch', {
    body: { ids: [enqNeeds.body.id, enqPassed.body.id], approve: true },
  });
  assert.equal(batch.status, 200);
  // needs_review item: not executed, still queued, surfaced in skipped[]
  assert.ok(batch.body.skipped.some((x) => x.id === enqNeeds.body.id), 'needs_review must be skipped');
  const needsAfter = await callAds('GET', '/api/v1/store/ads/action-queue/' + enqNeeds.body.id);
  assert.equal(needsAfter.body.state, 'queued');
  // passed item executed
  const passedAfter = await callAds('GET', '/api/v1/store/ads/action-queue/' + enqPassed.body.id);
  assert.equal(passedAfter.body.state, 'executed');
});

// M3-P0-04: raw forged body.guardrail.status='passed' + body.state='approved'
// must NOT be trusted — server recomputes guardrail and forces state='queued'.
test('M3-P0-04: enqueueManualAction ignores forged guardrail/state', async () => {
  const enq = await callAds('POST', '/api/v1/store/ads/action-queue', {
    body: {
      sourceStrategyName: 'LX manual operation',
      entity: { kind: 'campaign', id: 'cmp-forge-' + Date.now(), name: 'forge' },
      typedAction: { actionPrimitive: 'TOGGLE_CAMPAIGN', sourceSurface: 'lx', currentValue: { enabled: true }, recommendedValue: { enabled: false } },
      guardrail: { status: 'passed', reasons: ['forged'] },
      state: 'approved',
    },
  });
  assert.ok([200, 201].includes(enq.status));
  assert.equal(enq.body.state, 'queued', 'forged state=approved must be ignored');
  assert.notEqual(enq.body.guardrail.status, 'passed', 'forged guardrail=passed must be recomputed away');
  // direct execute must be blocked by approval_required (not a queued-passed bypass)
  const exec = await callAds('POST', '/api/v1/store/ads/action-queue/' + enq.body.id + '/execute', { body: {} });
  assert.equal(exec.status, 400);
  assert.match(exec.body.message, /approval_required/);
});

  const sid = s.body.id;
  const cmps = (await callAds('GET', '/api/v1/store/ads/lx/campaigns')).body.items.slice(0, 2);
  const r = await callAds('POST', '/api/v1/store/ads/strategies/' + sid + '/bind', { body: { campaignIds: cmps.map((c) => c.id) } });
  assert.equal(r.body.bindings.length, 2);

  const inv = await callAds('GET', '/api/v1/store/ads/lx/campaigns/' + cmps[0].id + '/strategies');

  // Unbind one
  const r2 = await callAds('POST', '/api/v1/store/ads/strategies/' + sid + '/bind', { body: { campaignIds: [cmps[1].id] } });
  assert.equal(r2.body.bindings.length, 1);
  // bindingsCount mirror
  const fresh = await callAds('GET', '/api/v1/store/ads/strategies/' + sid);
  assert.equal(fresh.body.bindingsCount, 1);
  globalThis.__qaM2mStratId = sid;
  globalThis.__qaM2mCmpIds = cmps.map((c) => c.id);
});

async function lastLog(actionType) {
  const db = getDbInstance();
  const row = db.prepare(
    `SELECT * FROM audit_logs
     WHERE user_id=? AND store_id=? AND action_type=?
     ORDER BY executed_at DESC, rowid DESC LIMIT 1`
  ).get(USER_ID, STORE_ID, actionType);
  if (!row) return null;
  return {
    id: row.id,
    userId: row.user_id,
    storeId: row.store_id,
    sourceModule: row.source_module,
    actionType: row.action_type,
    resourceType: row.resource_type,
    resourceId: row.resource_id,
    status: row.status,
    reverted: row.reverted === 1 || row.reverted === true,
    revertedAt: row.reverted_at,
